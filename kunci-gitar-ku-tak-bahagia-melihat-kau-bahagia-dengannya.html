f(x) = 4x + 4x - 3x

f(10) = 4(10) + 4(10) - 3(10)

f(10) = ( 4 × 10 ) + ( 4 × 10 ) - ( 3 × 10 )

f(10) = 40 + 40 - 30

f(10) = 80 - 30

f(10) = 50
