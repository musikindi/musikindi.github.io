1. diketahui : a + b = 37
a - b = 7
a + b = 37
a = 37 - b
substitusikan
a - b = 7
(37 - b) - b = 7
37 - b - b = 7
37 - 2b = 7
-2b = 7 - 37
-2b = -30
b = -30/-2
b = 15

a - b = 7
a - 15 = 7
a = 7 + 15
a = 22

a x b = 22 x 15 = 330

2. diketahui : R + 3J = 32.500
2R + 4J = 50.000
R + 3J = 32.500
R = 32.500 - 3J
substitusi kan
2R + 4J = 50.000
2(32.500 - 3J) + 4J = 50.000
65.000 - 6J + 4J = 50.000
65.000 - 2J = 50.000
-2J = 50.000 - 65.000
-2J = -15.000
J = -15.000 : -2

R = 32.500 - 3J
= 32.500 - 3(7.500)
= 32.500 - 22.500
= 10.000
